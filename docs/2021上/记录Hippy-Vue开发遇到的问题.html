function getActions(circleNumbers) {
  const actions = [];
  const unitDelay = 220;

  for (let i = 0; i < circleNumbers; i++) {
    actions.push({
      opacity: {
        startValue: 0,
        toValue: 1,
        duration: circleNumbers * unitDelay,
        delay: unitDelay * i,
        repeatCount: -1,
      },
    });
  }
  return actions;
}

export default {
  mounted() {
    this.onGetActions();
  },
  methods: {
    onGetActions() {
      const { circleNumbers } =  this;
      const actions = getActions(circleNumbers);
      this.actions = actions;
    },
  },
}
